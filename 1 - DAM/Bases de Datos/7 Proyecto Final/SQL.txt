CREATE TABLE PEDIDO(
CODIGO VARCHAR(20) NOT NULL,
DNI_CLIENTE VARCHAR(9) NOT NULL,
FECHA DATE NOT NULL,
CANTIDAD NUMBER(2) NOT NULL,
PRECIO NUMBER(6, 2) NOT NULL,
PRIMARY KEY (CODIGO),
FOREIGN KEY (DNI_CLIENTE) REFERENCES CLIENTE(DNI)
);

CREATE TABLE PRODUCTO(
CODIGO VARCHAR(20) NOT NULL,
NOMBRE VARCHAR(50) NOT NULL,
EXPLICACION VARCHAR(1000) NOT NULL,
PRECIO NUMBER(6, 2) NOT NULL,
PRIMARY KEY (CODIGO)
);

CREATE TABLE PEDIDO_PRODUCTO(
CODIGO_PEDIDO VARCHAR(20) NOT NULL,
CODIGO_PRODUCTO VARCHAR(20) NOT NULL,
PRIMARY KEY (CODIGO_PEDIDO, CODIGO_PRODUCTO),
FOREIGN KEY (CODIGO_PEDIDO) REFERENCES PEDIDO(CODIGO),
FOREIGN KEY (CODIGO_PRODUCTO) REFERENCES PRODUCTO(CODIGO)
);

CREATE TABLE CLIENTE(
DNI VARCHAR(9) NOT NULL,
NOMBRE VARCHAR(50) NOT NULL,
APELLIDO1 VARCHAR(50) NOT NULL,
APELLIDO2 VARCHAR(50) NOT NULL,
FECHA_NAC DATE NOT NULL,
CORREO VARCHAR(100) NOT NULL,
PRIMARY KEY (DNI)
);

CREATE TABLE ENCARGO(
CODIGO VARCHAR(20) NOT NULL,
DNI_CLIENTE VARCHAR(9) NOT NULL,
DETALLES VARCHAR(1000) NOT NULL,
FECHA_LIMITE DATE NOT NULL,
PRECIO NUMBER(6, 2) NOT NULL,
PRIMARY KEY (CODIGO),
FOREIGN KEY (DNI_CLIENTE) REFERENCES CLIENTE(DNI)
);

CREATE TABLE PROYECTO(
CODIGO VARCHAR(20) NOT NULL,
NOMBRE VARCHAR(50) NOT NULL,
PRIMARY KEY (CODIGO)
);

CREATE TABLE EMPLEADO(
DNI VARCHAR(9) NOT NULL,
CODIGO_ENCARGO VARCHAR(20),
CODIGO_PROYECTO VARCHAR(20),
NOMBRE VARCHAR(50) NOT NULL,
APELLIDO1 VARCHAR(50) NOT NULL,
APELLIDO2 VARCHAR(50) NOT NULL,
FECHA_NAC DATE NOT NULL,
DIRECCIÓN VARCHAR(200) NOT NULL,
CORREO VARCHAR(100) NOT NULL,
PRIMARY KEY (DNI),
FOREIGN KEY (CODIGO_ENCARGO) REFERENCES ENCARGO(CODIGO),
FOREIGN KEY (CODIGO_PROYECTO) REFERENCES PROYECTO(CODIGO)
);

CREATE TABLE PRESENTACION(
CODIGO VARCHAR(20) NOT NULL,
EXPLICACION VARCHAR(1000) NOT NULL,
FECHA_PRESENTACION DATE NOT NULL,
FECHA_SALIDA DATE NOT NULL,
PRIMARY KEY (CODIGO)
);

CREATE TABLE EMPLEADO_PROYECTO_PRESENTACION(
DNI_EMPLEADO VARCHAR(9) NOT NULL,
CODIGO_PRESENTACION VARCHAR(20) NOT NULL,
CODIGO_PROYECTO VARCHAR(20) NOT NULL,
PRIMARY KEY (DNI_EMPLEADO, CODIGO_PRESENTACION),
FOREIGN KEY (DNI_EMPLEADO) REFERENCES EMPLEADO(DNI),
FOREIGN KEY (CODIGO_PRESENTACION) REFERENCES PRESENTACION(CODIGO),
FOREIGN KEY (CODIGO_PROYECTO) REFERENCES PROYECTO(CODIGO)
);
